Pattern 1

* * * * *
* * * * *
* * * * *
* * * * *
* * * * *
  • Python
  • C
  • Java

for i in range(1,5+1):
    for j in range(1,6):
        print('*', end='')
    print('')



#include<stdio.h>
main(){
	int i,j;
	for(i = 1; i <= 5; i++){
		for(j = 1; j <= 5; j++){
			printf("*");
		}
		printf("\n");
	}
}



class Pattern01{
	public static void main(String args[]){
		for(int i = 1; i <= 5; i++) {
            for(int j = 1; j <= 5; j++){
                        System.out.print("*");
            }
            System.out.println("");
		}
	}
}

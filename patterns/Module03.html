Pattern 24

*
* *
* * *
* * * *
* * * * *
  • Python
  • C
  • Java

n = 5
for i in range(1,n+1):
    for j in range(1, n-i+1):
        print(end=" ")
    for k in range(1, i+1):
        print("*", end = '')
    print()



#include<stdio.h>
main(){
	int i,j,k;
	for(i = 1; i <= 5; i++){
		for(j = 5; j >= i+1; j--){
			printf(" ");
		}
		for(k = 1; k <= i; k++){
			printf("*");
		}
		printf("\n");
	}
}



class Pattern24{
	public static void main(String args[]){
		for(int i = 1; i <= 5; i++) {
       			for(int j = 5; j >= i+1; j--){
		              		System.out.print(" ");
			}
			for(int k = 1; k <= i; k++){
		              		System.out.print("*");
			}
			System.out.println("");
		}

	}
}
